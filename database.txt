drop table speldb.playedgame;
create table if not exists speldb.playedgame
(
    playedgameid int not null auto_increment,
    nightid int not null,
    gameid int not null,
    gid int not null,
    p1point float,
    p2point float,
    p3point float,
    p4point float,
    p5point float,
    constraint playedgame_game_fkey
           foreign key(gameid) references game(gameid),
    constraint playedgame_night_fkey
           foreign key(nightid) references night(nightid),
    primary key(playedgameid)
);

           foreign key(seasonid) references speldb.season(seasonid),
    constraint night_host_fkey
           foreign key(playerid) references player(playerid),
