create table if not exists speldb.playedgame
(
    playedgameid int not null auto_increment,
    date date not null,
    seasonid int not null,
    hostid int not null,
    gameid int not null,
    gid int not null,
    p1pos int,
    p1point float,
    p2pos int,
    p2point float,
    p3pos int,
    p3point float,
    p4pos int,
    p4point float,
    p5pos int,
    p5point float,
    constraint playedgame_season_fkey
           foreign key(seasonid) references speldb.season(seasonid),
    constraint playedgame_host_fkey
           foreign key(hostid) references player(playerid),
    constraint playedgame_game_fkey
           foreign key(gameid) references game(gameid),
    primary key(playedgameid)
);
